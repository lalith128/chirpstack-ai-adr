from sklearn.linear_model import LinearRegression

# 1. Load Data
df = pd.read_csv(DATA_PATH)
df['Timestamp'] = pd.to_datetime(df['Timestamp'])
df = df.sort_values('Timestamp').reset_index(drop=True)

# 2. NEW FEATURE: Volatility (The Clue for Uncertainty)
# Calculates how unstable the signal has been over the last 3 packets
df['RSSI_Vol'] = df['RSSI'].rolling(3).std().fillna(0)
df['SNR_Vol'] = df['SNR'].rolling(3).std().fillna(0)

# 3. Physics Feature & Lags
df['PathLoss_Est'] = -20 * np.log10(df['Frequency'] / 868e6) - 0.05 * df['Hum_BME']
df['Prev_RSSI'] = df['RSSI'].shift(1)
df['Prev_SNR'] = df['SNR'].shift(1)
df = df.dropna().reset_index(drop=True)

# --- PHYSICS DISCOVERY ---
reg_hum = LinearRegression().fit(df[['Hum_BME']], df['RSSI'])
hum_slope = -0.1 if reg_hum.coef_[0] > -0.01 else reg_hum.coef_[0]

reg_temp = LinearRegression().fit(df[['Temp_BME']], df['SNR'])
temp_slope = -0.05 if reg_temp.coef_[0] > -0.01 else reg_temp.coef_[0]

reg_pres = LinearRegression().fit(df[['Pres_BME']], df['RSSI'])
pres_slope = 0.05 if reg_pres.coef_[0] < 0 else reg_pres.coef_[0]

print(f"Physics: HumSlope={hum_slope:.3f}, TempSlope={temp_slope:.3f}")

# --- CONTEXT-AWARE AUGMENTATION ---

# 1. Moisture (High Hum -> Drop RSSI + Medium Noise)
aug_humid = df.copy()
added_hum = np.random.uniform(10, 20, size=len(aug_humid))
aug_humid['Hum_BME'] += added_hum
aug_humid['RSSI'] += added_hum * hum_slope 
aug_humid['RSSI'] += np.random.normal(0, 1.0, size=len(aug_humid)) # Sigma ~ 1.0

# 2. Storm (Low Pres -> Drop RSSI + HIGH Noise)
aug_storm = df.copy()
dropped_pres = np.random.uniform(10, 20, size=len(aug_storm))
aug_storm['Pres_BME'] -= dropped_pres
aug_storm['RSSI'] -= dropped_pres * pres_slope 
aug_storm['RSSI'] -= np.random.normal(0, 2.5, size=len(aug_storm)) # Sigma ~ 2.5 (Distinguishable!)

# 3. EDGE OF CELL (The "Smart" Crash)
# Instead of random, we pick rows where signal was already weak/unstable
aug_edge = df[df['Prev_RSSI'] < -100].copy() # Only weak signals
if len(aug_edge) > 0:
    # We make them CRASH HARD
    aug_edge['RSSI'] -= np.random.uniform(5, 10, size=len(aug_edge))
    # We add MASSIVE noise
    aug_edge['RSSI'] += np.random.normal(0, 4.0, size=len(aug_edge)) # Sigma ~ 4.0
else:
    aug_edge = df.copy() # Fallback

# Combine
df_augmented = pd.concat([df, aug_humid, aug_storm, aug_edge], ignore_index=True)
df_augmented = df_augmented.sample(frac=1, random_state=42).reset_index(drop=True)

# Inputs
input_context = Input(shape=(X_c_train.shape[1],), name='Context')
input_env = Input(shape=(X_e_train.shape[1],), name='Env')
input_action = Input(shape=(X_a_train.shape[1],), name='Action')
input_target = Input(shape=(2,), name='Target_Input') 

merged = Concatenate()([input_context, input_env, input_action])

# Shared Body
x = Dense(128, activation='relu')(merged)
x = BatchNormalization()(x)
x = Dropout(0.2)(x)
x = Dense(64, activation='relu')(x)
x = BatchNormalization()(x)
x = Dropout(0.1)(x)

# Mean Head
mu_dense = Dense(32, activation='relu')(x)
output_mu = Dense(2, name='mu_output')(mu_dense)

# Uncertainty Head
var_dense = Dense(32, activation='relu')(x)
# FIX: Change bias to -1.0 (Sigma ~0.60). 
# This helps the model adapt to the variable noise (0.5 to 4.0) faster.
output_log_var_raw = Dense(2, name='log_var_raw', 
                           bias_initializer=tf.keras.initializers.Constant(-1.0))(var_dense)

def clip_log_var(x):
    return tf.clip_by_value(x, -5.0, 3.0) # Allow slightly higher sigma for crashes

output_log_var = tf.keras.layers.Lambda(clip_log_var, name='log_var_clamped')(output_log_var_raw)

# Loss Layer
class NLLLossLayer(tf.keras.layers.Layer):
    def call(self, inputs):
        y_true, mu, log_var = inputs
        precision = tf.exp(-log_var)
        loss = tf.reduce_mean(0.5 * (log_var + precision * tf.square(y_true - mu)))
        self.add_loss(loss)
        return mu

training_output = NLLLossLayer()([input_target, output_mu, output_log_var])

# Models
train_model = Model(inputs=[input_context, input_env, input_action, input_target], outputs=training_output)
infer_model = Model(inputs=[input_context, input_env, input_action], outputs=[output_mu, output_log_var])

# Predict
mu_pred_scaled, log_var_pred_scaled = infer_model.predict([X_c_test, X_e_test, X_a_test])

# Unscale Mean
mu_real = scaler_target.inverse_transform(mu_pred_scaled)
y_real = scaler_target.inverse_transform(y_test)

# CORRECT SIGMA UNSCALING
# Multiply by variance scale, do not divide
sigma_pred_scaled = np.exp(0.5 * log_var_pred_scaled)
scale_factors = np.sqrt(scaler_target.var_) 
sigma_real = sigma_pred_scaled * scale_factors 

# Check Correlation
errors_rssi = np.abs(y_real[:, 0] - mu_real[:, 0])
uncertainty_rssi = sigma_real[:, 0]
corr = np.corrcoef(errors_rssi, uncertainty_rssi)[0, 1]

# Plot
plt.figure(figsize=(12, 5))
plt.subplot(1, 2, 1)
indices = np.arange(50)
plt.errorbar(indices, mu_real[:50, 0], yerr=1.96*sigma_real[:50, 0], fmt='o', alpha=0.5)
